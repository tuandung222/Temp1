% Header/footer + numbering customisation for lecture PDFs.
% Loaded via pandoc --include-in-header.

\usepackage{fancyhdr}
\usepackage{lastpage}
\usepackage{titlesec}

% Source markdown already includes manual numbering (e.g. "# 3.2 Verification ...").
% Disable LaTeX article-class auto-numbering so headings are not double-numbered.
% Sections still appear in the TOC (controlled by tocdepth, not secnumdepth).
\setcounter{secnumdepth}{-1}
\setcounter{tocdepth}{3}

% Activate fancy page style globally
\pagestyle{fancy}
\fancyhf{}                              % clear default header/footer
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\footrulewidth}{0.2pt}

% Header: left = document title, right = current section
\fancyhead[L]{\small\nouppercase{\leftmark}}
\fancyhead[R]{\small\nouppercase{\rightmark}}

% Footer: center = Page X of Y
\fancyfoot[C]{\small Trang \thepage{} / \pageref{LastPage}}

% Plain page style (used on first page of chapters / TOC) also gets page numbers
\fancypagestyle{plain}{%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0pt}%
  \renewcommand{\footrulewidth}{0.2pt}%
  \fancyfoot[C]{\small Trang \thepage{} / \pageref{LastPage}}%
}

% Make \leftmark show section name (not chapter, since pandoc uses article-style)
% and \rightmark show subsection name. In article class, only \rightmark is set
% by section, so we override both.
\renewcommand{\sectionmark}[1]{\markboth{#1}{}}
\renewcommand{\subsectionmark}[1]{\markright{#1}{}}

% Tighten section title spacing slightly (long lectures benefit)
\titlespacing*{\section}{0pt}{1.6ex plus 0.5ex minus .2ex}{1.0ex plus .2ex}
\titlespacing*{\subsection}{0pt}{1.2ex plus 0.4ex minus .2ex}{0.8ex plus .2ex}
